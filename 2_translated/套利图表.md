/**
 * Calculate Simple Moving Average (SMA)
 * @param {Array<number>} data - Input data array
 * @param {number} period - SMA period
 * @returns {Array<number>} - Array containing SMA values
 */
function calculateSMA(data, period) {
    let sma = [];
    let sum = 0;

    for (let i = 0; i < data.length; i++) {
        sum += data[i];

        if (i >= period - 1) {
            sma.push(sum / period);
            sum -= data[i - period + 1];
        } else {
            sma.push(null); // If there are not enough data points to calculate SMA, return null
        }
    }

    // Fill null values
    let firstNonNull = sma.find(value => value !== null);
    for (let i = 0; i < sma.length; i++) {
        if (sma[i] === null) {
            sma[i] = firstNonNull;
        }
    }

    return sma;
}

/**
 * Calculate Standard Deviation (Stdev)
 * @param {Array<number>} data - Input data array
 * @param {number} period - Stdev calculation period
 * @returns {Array<number>} - Array containing Stdev values
 */
function calculateStdev(data, period) {
    let stdev = [];

    for (let i = 0; i < data.length; i++) {
        if (i >= period - 1) {
            let subset = data.slice(i - period + 1, i + 1);
            let mean = subset.reduce((a, b) => a + b, 0) / period;
            let variance = subset.reduce((a, b) => a + Math.pow(b - mean, 2), 0) / period;
            stdev.push(Math.sqrt(variance));
        } else {
            stdev.push(null); // If there are not enough data points to calculate Stdev, return null
        }
    }

    // Fill null values
    let firstNonNull = stdev.find(value => value !== null);
    for (let i = 0; i < stdev.length; i++) {
        if (stdev[i] === null) {
            stdev[i] = firstNonNull;
        }
    }

    return stdev;
}
