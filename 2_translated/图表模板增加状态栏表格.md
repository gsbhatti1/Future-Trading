function CreateObjectString(cols, rows) {
    var i = cols; // Column
    var j = rows; // Row
    var firstCols = 'a';
    var charValue = 0;
    var firstName = ""; // Member name of the object string
    var lastName = "";  // Number part of the member name of the object string
    var strMember = ""; // Member of the object string
    var objStr = "";// Returned string
    if(i > 26){
        throw "ERROR: column must be less than 26";
    }
    var strHead = '{';
    var strEnd = '}';
    for(var n = 0 ; n < j; n++){
        // Process rows
        for(var m = 0 ; m < i; m++){
            // Process columns, handling 'abc', number=n
            charValue = firstCols.charCodeAt(); // Get the character encoding
            firstName = String.fromCharCode(charValue + m);
            lastName = n;
            if(n === j - 1 && m === i - 1){
                strMember = '"' + firstName + lastName + '"' + ':' + '""';
            }else{
                strMember = '"' + firstName + lastName + '"' + ':' + '""' + ',';
            }
            objStr += strMember;
        }
    }
    objStr = strHead + objStr + strEnd;
    return objStr;
}
