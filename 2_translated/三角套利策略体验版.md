var initAccount={//Initial account
BTC:0.009,
LTC:0.7,
USDT:55,
}

var doAccount={//Operation account initialization
BTC:0.009,
LTC:0.7,
USDT:55,
}

var tax = 0.0015;//Transaction rate, 0.15%

function getBaseData(){
var P1, P2, P3; //Basic price data, respectively btc_usdt, ltc_usdt, ltc_btc trading pairs
P1 = exchanges[0].GetTicker();
P2 = exchanges[1].GetTicker();
P3 = exchanges[2].GetTicker();
return [P1,P2,P3];
}

function main() {
Log(initAccount);
var xunhuan = true;//Whether the program continues execution is used to determine whether the account balance satisfies the operation, which is not considered here.
while(xunhuan){
var baseData = getBaseData();//Get basic data
if((baseData[1].Buy-baseData[0].Sell*baseData[2].Sell)*Q3>Q3*tax*(2*baseData[1].Buy+baseData[0].Sell*baseData[2].Sell)){//Considering the arbitrage conditions of handling fees (P2-P1*P3)*Q3>Q3*tax*(2*P2+P1*P3), please note that in order to ensure that all orders can be completed, the counterparty price is used for judgment. In actual operations, the depth of the transaction also needs to be considered.
if(xunhuan){
doAccount.LTC = doAccount.LTC + Q3*(1-tax);
doAccount.BTC = doAccount.BTC - baseData[2].Sell*Q3;//Buy LTC, sell BTC
doAccount.LTC = doAccount.LTC - Q3;
doAccount.USDT = doAccount.USDT + baseData[1].Buy*Q3*(1-tax);//Sell LTC and get USDT
doAccount.BTC = doAccount.BTC + baseData[2].Sell*Q3*(1-tax);
doAccount.USDT = doAccount.USDT - baseData[0].Sell*baseData[2].Sell*Q3;//Buy BTC and get USDT
Log('Usdt income:',doAccount.USDT-initAccount.USDT,'usdt','Positive cycle arbitrage','#FF0000');
Log('Real income:',doAccount.USDT-initAccount.USDT-(initAccount.LTC-doAccount.LTC)*baseData[1].Buy-(initAccount.BTC-doAccount.BTC)*baseData[2].Sell,'#0000FF')//Real income after deducting handling fees
Log(doAccount.USDT-initAccount.USDT-(initAccount.LTC-doAccount.LTC)*baseData[1].Buy-(initAccount.BTC-doAccount.BTC)*baseData[2].Sell,'#0000FF')
}else{
xunhuan = false;
}
}else if(baseData[0].Buy*baseData[2].Buy-baseData[1].Sell>(2*baseData[0].Buy*baseData[2].Buy+baseData[1].Sell)*tax){//Considering the reverse cycle arbitrage of handling fees
if(xunhuan){
doAccount.LTC = doAccount.LTC - Q3;
doAccount.BTC = doAccount.BTC + baseData[2].Buy*Q3*(1-tax);
doAccount.LTC = doAccount.LTC + Q3*(1-tax);
doAccount.USDT = doAccount.USDT - baseData[1].Sell*Q3;
doAccount.BTC = doAccount.BTC - baseData[2].Buy*Q3;
doAccount.USDT = doAccount.USDT + baseData[0].Buy*baseData[2].Buy*Q3*(1-tax);
Log('Usdt income:',doAccount.USDT-initAccount.USDT,'usdt,reverse cycle arbitrage','#FF0000');
Log('Real income:',doAccount.USDT-initAccount.USDT-(initAccount.LTC-doAccount.LTC)*baseData[1].Sell-(initAccount.BTC-doAccount.BTC)*baseData[2].Buy,'#0000FF')
Log(doAccount.USDT-initAccount.USDT-(initAccount.LTC-doAccount.LTC)*baseData[1].Buy-(initAccount.BTC-doAccount.BTC)*baseData[2].Sell,'#0000FF')
}else{
xunhuan = false;
}
}
Sleep(1000);//The program sleeps for 1 second
}
}
